import { Input, Box, Button, Grid, Label } from "@theme-ui/components"
import { mix } from "@theme-ui/color"
import Prism from "@theme-ui/prism"
import CodeBlock from '../../../src/components/codeBlock'

I like this trick for using colors in user interfaces. Even a slight saturation makes difference. Try switching between different colors by clicking buttons below and see how the input changes:

export const Form = ({ children }) => {
  const [color, setColor] = React.useState("#0984e3")
    const colors = [
    {"label": "blue", "code": '#0984e3'},
    {"label": "green", "code": '#00a383'},
    {"label": "pink", "code": '#fd79a8'},
    {"label": "purple", "code": '#6c5ce7'},
  ]
  return (
    <>
      <Grid columns={[2, colors.length]} gap={1}>
        {colors.map(c => (
          <Button
            onClick={() => setColor(c.code)}
            color={color === c.code ? "background" : "text"}
            key={c.label}
            sx={{
          bg: color === c.code ? c.code : mix(c.code, "background", 0.1),
          color: color === c.code ? 'background' : mix(c.code, "text", 0.7)
          }}
          >
            {c.label}
          </Button>
        ))}
      </Grid>
      <Box
        as="form"
        sx={{
          bg: mix(color, "background", 0.05),
          padding: [3, 4],
          mt: 2,
          borderRadius: 5,
        }}
      >
        <Label sx={{ color: mix(color, "text", 0.3) }} htmlFor="username">
          Username
        </Label>
        <Input
          name="username"
          placeholder="Type something..."
          sx={{
            bg: "background",
            borderColor: mix(color, "background", 0.3),
            color: mix(color, "text", 0.35),
            outline: "none",
            ":focus": { borderColor: mix(color, "background", 0.8) },
            "::placeholder": {
              color: mix(color, "text", 0.5),
              opacity: 0.3,
            },
          }}
        />
      </Box>
      
      {children(color, colors)}
    </>
  )
}

<Form>
  {(color, colors) => (
    <CodeBlock className="jsx">
      {`
export default {
  colors: {
    primary: "${color}", // ${colors.filter(c => c.code === color).map(c => c.label)}
    text: "hsl(0, 0%, 10%)",
    background: "white",
  }
  forms {
    input {
      bg: "background",
      borderColor: mix(color, "background", 0.3),
      color: mix(color, "text", 0.35),
      outline: "none",
      ":focus": { borderColor: mix(color, "background", 0.8) },
      "::placeholder": {
        color: mix(color, "text", 0.5),
        opacity: 0.3
      }
    }
    label {
      color: mix(color, "text", 0.3)
    }
  }
}
`}
    </CodeBlock>
  )}
</Form>
